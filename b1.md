# 🧩 4. Bayesian Tree Model (pgmpy)

```python
from pgmpy.models import BayesianModel
from pgmpy.factors.discrete import TabularCPD

# Graph
model = BayesianModel([
    ('Domain', 'Route'),
    ('Intent', 'Route'),
])

# Domain CPD
cpd_domain = TabularCPD('Domain', 2, [[0.6], [0.4]])  
# 0=urs, 1=insurance

# Intent CPD
cpd_intent = TabularCPD('Intent', 2, [[0.5], [0.5]])  
# 0=generate, 1=validate

# Route CPD (initial routing policy)
cpd_route = TabularCPD(
    variable='Route',
    variable_card=4,
    values=[
        # D=0,I=0 | D=0,I=1 | D=1,I=0 | D=1,I=1
        [0.7, 0.1, 0.2, 0.1],  # urs_generator
        [0.2, 0.7, 0.1, 0.1],  # urs_validator
        [0.1, 0.1, 0.6, 0.2],  # knowledge_agent
        [0.0, 0.1, 0.1, 0.6],  # fallback
    ],
    evidence=['Domain', 'Intent'],
    evidence_card=[2, 2]
)

model.add_cpds(cpd_domain, cpd_intent, cpd_route)
```

---

# 🔮 5. Inference: Choose Agent Dynamically

```python
from pgmpy.inference import VariableElimination

infer = VariableElimination(model)

def route(domain, intent):
    q = infer.query(['Route'], evidence={'Domain': domain, 'Intent': intent})
    probs = q['Route'].values
    best = probs.argmax()
    return AGENTS[best], probs
```

---

### Example:

```python
agent, probs = route(domain=0, intent=0)
print(agent, probs)
```

# 🔄 6. Dynamic Learning (Online Update)

We maintain **success counters**.

```python
import numpy as np

success_matrix = np.ones((4, 4))  # Laplace smoothing
failure_matrix = np.ones((4, 4))

def update_policy(domain, intent, agent_id, success):
    idx = domain*2 + intent
    if success:
        success_matrix[agent_id, idx] += 1
    else:
        failure_matrix[agent_id, idx] += 1

    # Update CPD dynamically
    total = success_matrix[:, idx] + failure_matrix[:, idx]
    cpd_route.values[:, idx] = success_matrix[:, idx] / total
```

```python
import random

for step in range(100):
```
